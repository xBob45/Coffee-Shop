#apache/Docker/coffee-shop.conf
<VirtualHost *:443>
    ServerAdmin webmaster@localhost
    ServerName localhost
    DocumentRoot /var/www/html/
    #DirectoryListing-1 - START
    #DirectoryListing-1 - END
    ErrorLog ${APACHE_LOG_DIR}/error-ssl.log
    CustomLog ${APACHE_LOG_DIR}/access-ssl.log combined
                
    SSLEngine on
    SSLCertificateFile      /etc/apache2/ssl/server.crt
    SSLCertificateKeyFile /etc/apache2/ssl/server.key

    ProxyPass /app http://127.0.0.1:5000/
    ProxyPassReverse /app http://127.0.0.1:5000/

    # Mod_rewrite rules for URL structure
    RewriteEngine On

    # Redirect root URL to /app
    RewriteRule ^/$ /app [R=307,L]

    # Exclude certain paths from rewriting
    RewriteCond %{REQUEST_URI} !^/Coffee-Shop
    RewriteCond %{REQUEST_URI} !^/icons/
    RewriteCond %{REQUEST_URI} !^/app
    RewriteRule ^(.*)$ /app$1 [R=307,L]
    
</VirtualHost>
